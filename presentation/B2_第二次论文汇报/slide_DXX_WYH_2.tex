        \item 调查对象为3659 名儿童(6-19岁)和6566 名成人(20 岁);

        $$y_i\sim mTobit(\omega,\mu_i,\sigma^2,L=log(2.3))$$
        \begin{tiny}
        $$\mu_i=\beta_0+\beta_1Age_i+\beta_2Gender_i+\beta_3Race_i+\beta_4Edu_i+\beta_5BMI_i+\beta_6Cotinine_i+\beta_7Creatinine_i$$
        \end{tiny}
    \end{itemize}

\subsection[检验结果]{检验结果}\label{subsec:3-3}
\begin{frame}
	\frametitle{检验结果}
    \begin{itemize}
    \item 对于$H_0:\omega=0\:\:vs\:\:\omega>0$：
    \begin{itemize}
    \item Wald、LR、score 和new Test 分别代表的P值为：
    $$0.5000,0.5037,0.768, 0.986$$
    \end{itemize}
    \item 四个测试都得出了同样得结论，即没有显著证据表明能够拒绝$H_0$。
    \item 即尿中三氯生得浓度没有潜在得类别。

    \end{itemize}
\end{frame}
